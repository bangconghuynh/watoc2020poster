  \begin{column}{\colwidth}
    \begin{block}{1. The Exact Adiabatic Connection}
      For an $\gls*{gen:Ne}$-electron system, consider a $\lambda$-parametrised electronic Hamiltonian
      \begin{equation*}
        \gls*{op:hamil}[_{\lambda}](\gls*{op:pot1e}_{\lambda})
        =
        \gls*{op:kin}
        + \lambda \gls*{op:ee}
        + \sum_{i=1}^{\gls*{gen:Ne}} \gls*{op:pot1e}_{\lambda}(\gls*{bas:spatialcoord}[_i]),
        \quad
        \textrm{where}
        \quad
        \gls*{op:ee} = \sum_{i=1}^{\gls*{gen:Ne}}\sum_{j>i}^{\gls*{gen:Ne}} \frac{1}{\lvert \gls*{bas:spatialcoord}[_i] - \gls*{bas:spatialcoord}[_j] \rvert},
      \end{equation*}
      that continuously links the \emphbold{LightSalmon}{physical system} to the \emphbold{LightSeaGreen}{non-interacting system}.

      \tikzextrue
      \tikzexternalenable
      \begin{figure}
        \centering
        \useexternalfile{0.913}{0}{0}{acdiagram}
        \label{fig:acdiagram}
      \end{figure}
      \tikzexternaldisable
      \tikzexfalse

      The \emphbold{Red}{potential $\gls*{op:pot1e}_{\lambda}$} is chosen for $\gls*{op:hamil}[_{\lambda}]$ to admit the same \emphbold{Red}{ground density $\gls*{den:ground}$} as $\gls*{op:hamil}[_{1}]$ via the \emphbold{red}{ground density matrix $\gls*{op:denen}[_{\lambda}](\gls*{den:ground})$}.

      The \emphbold{Blue}{adiabatic connection} is defined by $\symcal{W}_{\lambda}(\gls*{den:ground}) = \tr \gls*{op:denen}[_{\lambda}](\gls*{den:ground}) \gls*{op:ee}$, which we calculate accurately for small systems to form approximate models for larger systems.
    \end{block}
  \end{column}

  \separatorcolumn

  \begin{column}{\colwidth}
    \begin{block}{2. The Lieb Variational Principle}
      Consider the space of densities $\gls*{struct:denspace}$ and its dual, the space of potentials $\gls*{struct:potspace}$:
      \begin{equation*}
        \gls*{struct:denspace} = L^{3}(\symbb{R}^3) \cap L^{1}(\symbb{R}^3),\quad
        \gls*{struct:potspace} = L^{3/2}(\symbb{R}^3) + L^{\infty}(\symbb{R}^3).
      \end{equation*}
      For every $\lambda$ along the adiabatic connection, given a \emphbold{Blue}{wavefunction method M} to compute the \emphbold{Red}{M-ground energy $E^{\symup{M}}_{\lambda}(\gls*{op:pot1e})$} of $\gls*{op:hamil}[_{\lambda}](\gls*{op:pot1e})$ and the M-reference density $\gls*{den:ground}[^{\symup{M}}_1] \in \gls*{struct:denspace}$, by finding the \emphbold{Red}{Lieb universal functional}
      \begin{equation*}
        F^{\symup{M}}_{\lambda}(\gls*{den:ground}[^{\symup{M}}_1])
          = \sup_{\gls*{op:pot1e} \in \gls*{struct:potspace}}
          \,\left[%
            E^{\symup{M}}_{\lambda}(\gls*{op:pot1e})
            - \int \gls*{op:pot1e}(\gls*{bas:spatialcoord}) \gls*{den:ground}[^{\symup{M}}_1](\gls*{bas:spatialcoord}) \ \symup{d} \gls*{bas:spatialcoord}%
          \right]
          \equiv \sup_{\gls*{op:pot1e} \in \gls*{struct:potspace}}
          G^{\symup{M}}_{\lambda}(\gls*{op:pot1e}; \gls*{den:ground}[^{\symup{M}}_1]),
      \end{equation*}
      one obtains the \emphbold{Red}{M-optimal potential $\gls*{op:pot1e}^{\symup{M}}_{\lambda}$} that supports $\gls*{den:ground}^{\symup{M}}_{\lambda}$ as its ground density.

      The higher the quality of the wavefunction method M:
      \begin{itemize}
        \item the better $\gls*{den:ground}[^{\symup{M}}_1]$ approximates the true reference density $\gls*{den:ground}$;
        \item the lower the Lieb-variational errors $\lVert \gls*{den:ground}^{\symup{M}}_{\lambda} - \gls*{den:ground}[^{\symup{M}}_1] \rVert_p$ where $1 \le p \le 3$; and
        \item the better $\gls*{op:pot1e}^{\symup{M}}_{\lambda}$ approximates the true potential $\gls*{op:pot1e}_{\lambda}$.
      \end{itemize}

    \end{block}
  \end{column}

    \begin{alertblock}{3. Challenges of Degenerate Systems}
      Consider a Hamiltonian $\gls*{op:hamil}[_{\lambda}](\gls*{op:pot1e})$ having a \emphbold{Red}{symmetry group $\gls*{struct:gengroup}_{\lambda}(\gls*{op:pot1e})$} and admitting a \emphbold{Blue}{degenerate ground wavefunction $\gls*{wf:ground}[^{\symup{M}}_{\lambda}](\gls*{op:pot1e})$} with \emphbold{Red}{energy $E^{\symup{M}}_{\lambda}(\gls*{op:pot1e})$} and \emphbold{Red}{density $\gls*{den:ground}[_{\lambda}^{\symup{M}}](\gls*{bas:spatialcoord}; \gls*{op:pot1e})$}.

      \begin{columns}[t]
        \begin{column}{\dimexpr(.65\colwidth)}
          \centering
          \begin{orangehighlightbox}[height=15cm, overlay, remember as=gradbox]{Well-defined gradient?}
            The Lieb optimisation procedure to obtain the M-optimal potential $\gls*{op:pot1e}^{\symup{M}}_{\lambda}$ requires the functional derivative
            \begin{equation*}
              \frac{%
                \delta G^{\symup{M}}_{\lambda}(\gls*{op:pot1e}; \gls*{den:ground}[^{\symup{M}}_1])%
              }{%
                \delta \gls*{op:pot1e}(\gls*{bas:spatialcoord})%
              }
              = \textcolor{Red}{%
                \frac{%
                  \delta E^{\symup{M}}_{\lambda}(\gls*{op:pot1e})%
                }{%
                  \delta \gls*{op:pot1e}(\gls*{bas:spatialcoord})%
                }%
              } - \gls*{den:ground}[^{\symup{M}}_1](\gls*{bas:spatialcoord}).
            \end{equation*}
            As $\gls*{den:ground}[_{\lambda}^{\symup{M}}](\gls*{bas:spatialcoord}; \gls*{op:pot1e})$ is invariant only under a proper subgroup of $\gls*{struct:gengroup}_{\lambda}(\gls*{op:pot1e})$, the functional derivative $\textcolor{Red}{%
              \delta E^{\symup{M}}_{\lambda}(\gls*{op:pot1e}) / \delta \gls*{op:pot1e}(\gls*{bas:spatialcoord})%
            }$ is \emphbold{Blue}{not unique}:
            \begin{equation*}
              \textcolor{Red}{%
                \frac{%
                  \delta E^{\symup{M}}_{\lambda}(\gls*{op:pot1e})%
                }{%
                  \delta \gls*{op:pot1e}(\gls*{bas:spatialcoord})%
                }%
              } =
              \sum_{i=1}^{\lvert\gls*{struct:gengroup}_{\lambda}(\gls*{op:pot1e})\rvert}
                c_i
                \hat{g}_i
                \gls*{den:ground}[_{\lambda}^{\symup{M}}](\gls*{bas:spatialcoord}; \gls*{op:pot1e}),
              \quad
              g_i \in \gls*{struct:gengroup}_{\lambda}(\gls*{op:pot1e}),
              \ c_i \ge 0,
              \ \sum_{i=1}^{\lvert\gls*{struct:gengroup}_{\lambda}(\gls*{op:pot1e})\rvert} c_i = 1.
            \end{equation*}
          \end{orangehighlightbox}
        \end{column}

        \begin{column}{\dimexpr(.65\colwidth)}
          \centering
          \begin{violethighlightbox}[height=15cm, overlay, remember as=sympotbox]{Symmetry of potential?}
            At all $\lambda$, the M-optimal potential $\gls*{op:pot1e}[^{\symup{M}}_{\lambda}]$ determines the symmetry group of the Hamiltonian:
            \begin{alignat*}{4}
              \lambda = 1:   &\quad \gls*{op:pot1e}[_{\symup{ext}}] &&\mapsto \gls*{struct:gengroup}_1 \quad &&\textrm{physical system},\\
              \lambda \ne 1: &\quad \gls*{op:pot1e}[^{\symup{M}}_{\lambda}] &&\mapsto \gls*{struct:gengroup}_{\lambda}(\gls*{op:pot1e}[^{\symup{M}}_{\lambda}]) \quad &&\textrm{auxiliary systems}.
            \end{alignat*}
            Is \emphbold{Blue}{equality} in the following condition
            \begin{equation*}
              \gls*{struct:gengroup}_1 \ge \gls*{struct:gengroup}_{\lambda}(\gls*{op:pot1e}[^{\symup{M}}_{\lambda}]),
              \quad
              \lambda \ne 1
            \end{equation*}
            necessary for $\gls*{op:pot1e}[^{\symup{M}}_{\lambda}] \in \gls*{struct:potspace}$?
            In other words, should the \emphbold{Red}{auxiliary systems} \emphbold{Blue}{respect} the symmetry of the \emphbold{Red}{physical system}?
          \end{violethighlightbox}
        \end{column}

        \begin{column}{\dimexpr(.65\colwidth)}
          \centering
          \begin{greenhighlightbox}[height=15cm, overlay, remember as=vrepbox]{Guarantee of $\symbfit{v}$-representability?}
            The existence of the adiabatic connection $\symcal{W}_{\lambda}(\gls*{den:gen})$ requires the \emphbold{Red}{reference density $\gls*{den:gen}$} to be \emphbold{Red}{$v$-representable $\forall \lambda \in \interval{0}{1}$}.
            However, general $v$-representability conditions for $\gls*{den:gen} \in \gls*{struct:denspace}$ are not well-established.
            In particular:
            \begin{itemize}
              \item Is $\gls*{den:ground}[^{\symup{M}}_1]$ \emphbold{Red}{pure-state} $v$-representable $\forall \lambda \in \interval{0}{1}$?
              \item Is the ensemble density
              \begin{equation*}
                \gls*{den:ground}[^{\symup{M}}_{1, \symup{en}}] =
                \frac{1}{\lvert \gls*{struct:gengroup}_1 \rvert}
                \sum_{i=1}^{\lvert \gls*{struct:gengroup}_1 \rvert}
                \hat{g}_i \gls*{den:ground}[^{\symup{M}}_1]
              \end{equation*}
              \emphbold{Red}{ensemble-state} $v$-representable $\forall \lambda \in \interval{0}{1}$?
            \end{itemize}
          \end{greenhighlightbox}
        \end{column}
      \end{columns}
      \vspace{1cm}

    \end{alertblock}
